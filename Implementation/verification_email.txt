function verification_email() {
    var email = $('#email').val();

    $.ajax({
        type: 'POST',
        url: '../php/email.php',
        data: { email: email },
        success: function(result) {
            if (result === 'True') {
                var hrefL = "";
                window.location.replace(hrefL);
            }
        }
    });
}
